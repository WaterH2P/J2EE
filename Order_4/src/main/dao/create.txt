# create user account table
DROP TABLE IF EXISTS userInfo;
CREATE TABLE userInfo(
  userID INTEGER UNSIGNED ,
  password VARCHAR(30) NOT NULL ,
  PRIMARY KEY( userID )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

# create order info table
DROP TABLE IF EXISTS orderInfo;
CREATE TABLE orderInfo(
  orderID INTEGER UNSIGNED ,
  userID INTEGER UNSIGNED NOT NULL ,
  goodID INTEGER UNSIGNED NOT NULL ,
  goodNum INTEGER UNSIGNED NOT NULL ,
  totalPrice FLOAT UNSIGNED NOT NULL ,
  orderDate DATETIME NOT NULL ,
  PRIMARY KEY( orderID )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

# create good info table
DROP TABLE IF EXISTS goodInfo;
CREATE TABLE goodInfo(
  goodID INTEGER UNSIGNED ,
  goodName VARCHAR(30) NOT NULL ,
  goodPrice FLOAT UNSIGNED NOT NULL ,
  storeNum INTEGER UNSIGNED NOT NULL ,
  PRIMARY KEY( goodID )
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
